func jumpingBinarySearch(arr []int, x int) int {
	n := len(arr)
	k := 0

	for dist := n / 2; dist >= 1; dist /= 2 {
		for k+dist < n && arr[k+dist] <= x {
			k = k + dist
		}
	}
	if arr[k] == x {
		return k
	} else {
		return -1
	}

}
